### 指针和数组
```c
    int array[10];
    int* ap = array;
    int* ap2 = array + 2;
    for (int i = 0; i < 10; i++)
    {
        array[i] = i;
    }
    cout << "*(ap+i)  " << "*(ap2+i)  " << "*(ap+i-2)" << endl;
    for (int i = 0; i < 10; i++)
    {
        cout << *(ap + i) <<"          ";
        cout << *(ap2 + i) << "         ";
        cout << *(ap2 + i - 2) << endl;
    }
    cout << array[-10] << endl;
    cout << ap[-10] << endl;
    cout << ap2[-10] << endl;
    cout << array[100] << endl;

    cout << "*****地址******" << endl;
    for (int i = 0; i < 10; i++)
    {
        cout<<"array["<< i<<"]:   " << &array[i] << endl;
    }
    cout << (long long int)&array[1] - (long long int)&array[0] << endl;
    cout << &array[1] - &array[0] << endl;
    cout << &array[2] << endl;
    cout << &*(ap+2) << endl;
    cout << &*(ap2) << endl << endl;

    cout << ap << endl;
    cout << array << endl;
    cout << &ap << endl;//指针变量有地址，此表达式没有对应的array表达式
    cout << &array << endl;//和array相同
    cout << &array[0] << endl;
    cout << &ap2 << endl;
    
    cout << 2[array] << endl;//合法a[b]->*(a+b)=>*(a+(array))，但是不应该这么编写
```

#### 效率
假定下标和指针都是正确的，下标绝不会比指针更有效率，但指针有时会比下标更有效率。（C和指针）
却决于编译器和机器，也和编写的代码有关。

### 指针数组和数组指针
```c
    int array[10];
    int* ap = array;
    int* ap2 = array + 2;
    for (int i = 0; i < 10; i++)
    {
        array[i] = i;
    }
    //指针数组和数组指针
    int* array2[10];//指针数组
    int(*array3)[10];//数组指针
    for (int i = 0; i < 10; i++)
    {
        array2[i] = &array[i];
    }
    array3 = &array;
    for (int i = 0; i < 10; i++)
    {
        cout << array2[i]<< "   ";
        cout << *array2[i]<< endl;
    }
    for (int i = 0; i < 10; i++)
    {
        cout << (*array3)[i] << "   ";//所指向的数组中的元素
        cout << &(*array3)[i] << "   ";//所指向的数组中的元素 的地址
        cout << array3[i] << "   ";//首地址和数组首地址相同，每次地址偏移0x24->40->也就是一个数组内存大小。
        cout << *array3[i] << endl;//[]优先级高于*
    }
    //二维数组
    int array4[4][10];
    array3 = array4;
    for (int i = 0; i < 4; i++)
        for (int j = 0; j < 10; j++)
            array4[i][j] = 10 * i + j;
    for (int i = 0; i < 4; i++)
        for (int j = 0; j < 10; j++)
        {
            cout << *(*(array3+i)+j) << " ";
        }
    return 0;
```
